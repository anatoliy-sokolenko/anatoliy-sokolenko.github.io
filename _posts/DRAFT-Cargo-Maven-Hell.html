Despite the time Apache Maven has held leading place as build tool it is still extremely popular in convenient in many occasions.

There is however one area which is not well developed in Maven. It is execution of integration and functional tests. The issue is most of modern Java application require considerable amount of runtime dependencies and integration tests should not depend on particular environment. 

In following paper we will not describe implementation of CI process for complicated and multi-module applications. We will rather describe solution, how to create simple and fast build process for artifacts with 2-3 dependencies. 

Main idea of CI solution - to start 

<profiles>
    <profile>
      <id>with-functional-tests</id>

      <properties>
        <forfunctests.root.dir>${project.build.directory}/forfunc</forfunctests.root.dir>

        <forfunctests.jetty.version>7.6.3.v20120416</forfunctests.jetty.version>
        <forfunctests.jetty.dir>${forfunctests.root.dir}/jetty</forfunctests.jetty.dir>


        <forfunctests.fitnesse.version>20111026</forfunctests.fitnesse.version>
        <forfunctests.fitnesse.root.dir>${forfunctests.root.dir}/functional-tests-dist</forfunctests.fitnesse.root.dir>
      </properties>

      <!-- Extract functional tests -->
      <build>
        <plugins>
          <plugin>
            <artifactId>maven-dependency-plugin</artifactId>
            <version>2.6</version>

            <executions>
              <execution>
                <id>forfitnesse-explode-fitnesse-tests</id>
                <phase>package</phase>

                <goals>
                  <goal>unpack</goal>
                </goals>

                <configuration>
                  <artifactItems>
                    <artifactItem>
                      <groupId>me.sokolenko.maven-on-ci</groupId>
                      <artifactId>functional-tests</artifactId>
                      <version>${project.version}</version>
                      <type>zip</type>
                      <outputDirectory>${forfunctests.root.dir}</outputDirectory>
                    </artifactItem>
                    <artifactItem>
                      <groupId>me.sokolenko.maven-on-ci</groupId>
                      <artifactId>business-components</artifactId>
                      <version>${project.version}</version>
                      <type>war</type>
                      <outputDirectory>${forfunctests.jetty.dir}/httpstub</outputDirectory>
                    </artifactItem>
                  </artifactItems>
                </configuration>
              </execution>
            </executions>
          </plugin>

          <!-- Reserve TCP/IP ports for Jetty and FitNesse -->
          <plugin>
            <groupId>org.codehaus.mojo</groupId>
            <artifactId>build-helper-maven-plugin</artifactId>
            <version>1.7</version>
            <executions>
              <execution>
                <id>reserve-network-port</id>
                <phase>pre-integration-test</phase>
                <goals>
                  <goal>reserve-network-port</goal>
                </goals>
                <configuration>
                  <portNames>
                    <portName>forfunctests.jetty.port</portName>
                    <portName>forfunctests.jetty.stopPort</portName>
                  </portNames>
                </configuration>
              </execution>
            </executions>
          </plugin>

          <plugin>
            <groupId>org.codehaus.cargo</groupId>
            <artifactId>cargo-maven2-plugin</artifactId>
            <version>1.2.2</version>

            <configuration>
              <container>
                <containerId>jetty7x</containerId>

                <artifactInstaller>
                  <groupId>org.eclipse.jetty</groupId>
                  <artifactId>jetty-distribution</artifactId>
                  <version>${forfunctests.jetty.version}</version>
                </artifactInstaller>

                <systemProperties>
                  <saturntuning.config.location>${forfunctests.profile.path}</saturntuning.config.location>
                  <forfunctests.jetty.port>${forfunctests.jetty.port}</forfunctests.jetty.port>

                  <http.stub.mode>STUB</http.stub.mode>
                </systemProperties>
              </container>

              <configuration>
                <type>standalone</type>
                <home>${forfunctests.jetty.dir}/home</home>

                <properties>
                  <cargo.servlet.port>${forfunctests.jetty.port}</cargo.servlet.port>

                  <!-- This option is used to specify stopPort for Jetty -->
                  <cargo.rmi.port>${forfunctests.jetty.stopPort}</cargo.rmi.port>
                  <cargo.jvmargs>-Xmx1g -Xms1g -XX:MaxPermSize=256m</cargo.jvmargs>
                </properties>

                <deployables>
                  <deployable>
                    <groupId>com.macys.search.tuning</groupId>
                    <artifactId>saturn-tuning-webapp</artifactId>
                    <properties>
                      <context>/saturn-tuning</context>
                    </properties>
                    <pingURL>http://localhost:${forfunctests.jetty.port}/saturn-tuning</pingURL>
                    <pingTimeout>300000</pingTimeout> <!-- give it 5 minutes -->
                  </deployable>

                  <deployable>
                    <groupId>httpstub</groupId>
                    <artifactId>httpstub-web</artifactId>
                    <type>war</type>
                    <location>${forfunctests.jetty.dir}/httpstub</location>
                    <properties>
                      <context>/</context>
                    </properties>
                    <!-- HTTP Stub does not have any method which responds 200, except this one -->
                    <pingURL>http://localhost:${forfunctests.jetty.port}/stubconfig/mode?mode=stub</pingURL>
                    <pingTimeout>300000</pingTimeout> <!-- give it 5 minutes -->
                  </deployable>
                </deployables>
              </configuration>
            </configuration>
            <executions>
              <execution>
                <id>forfunctests-start-jetty</id>

                <phase>pre-integration-test</phase>

                <goals>
                  <goal>start</goal>
                </goals>
              </execution>
              <execution>
                <id>forfunctests-stop-jetty</id>
                <phase>post-integration-test</phase>

                <goals>
                  <goal>stop</goal>
                </goals>

                <configuration>
                  <container>
                    <containerId>jetty7x</containerId>
                  </container>
                </configuration>
              </execution>
            </executions>
          </plugin>
        </plugins>
      </build>
    </profile>
  </profiles>
