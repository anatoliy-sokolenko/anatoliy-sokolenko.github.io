<properties>
    <env>galak</env>
    <prepared.gwt.resources.directory>${project.build.directory}/prepared-gwt-resources</prepared.gwt.resources.directory>
</properties>

<build>
    <resources>
        <resource>
            <directory>${prepared.gwt.resources.directory}</directory>
        </resource>

        <resource>
            <directory>${basedir}/src/main/gwt-resources</directory>
            <targetPath>${prepared.gwt.resources.directory}</targetPath>
            <filtering>true</filtering>
        </resource>
        <resource>
            <directory>${basedir}/src/main/resources</directory>
            <filtering>true</filtering>
        </resource>
    </resources>

    <filters>
        <filter>${basedir}/src/main/config/${env}-config.properties</filter>
    </filters>

    <plugins>
        <plugin>
            <groupId>org.codehaus.mojo</groupId>
            <artifactId>gwt-maven-plugin</artifactId>
            <version>1.2</version>
            <configuration>
                <gwtVersion>${gwt.version}</gwtVersion>
                <disableCastChecking>true</disableCastChecking>
                <disableClassMetadata>true</disableClassMetadata>
                <logLevel>INFO</logLevel>
                <gwtVersion>${gwt.version}</gwtVersion>
                <extraJvmArgs>-Xmx512m</extraJvmArgs>
                <enableAssertions>${debug}</enableAssertions>

                <module>org.sokolenko.qa.gwt.QuestAssistant</module>

                <hostedWebapp>${project.build.directory}/hosted-webapp</hostedWebapp>
                <port>8080</port>
                <debugPort>8902</debugPort>
                <noServer>true</noServer>
                <runTarget>QuestAssistant.html</runTarget>
            </configuration>
            <executions>
                <execution>
                    <id>compile</id>
                    <phase>process-classes</phase>
                    <goals>
                        <goal>compile</goal>
                    </goals>
                </execution>
                <execution>
                    <id>generate async</id>
                    <phase>generate-sources</phase>
                    <goals>
                        <goal>generateAsync</goal>
                    </goals>
                </execution>
                <execution>
                    <id>debug ui</id>
                    <goals>
                        <goal>debug</goal>
                    </goals>
                </execution>
            </executions>
        </plugin>

        <plugin>
            <groupId>org.apache.maven.plugins</groupId>
            <artifactId>maven-war-plugin</artifactId>
            <configuration>
                <webResources>
                    <resource>
                        <directory>${basedir}/src/main/webapp</directory>
                        <filtering>true</filtering>
                    </resource>
                    <resource>
                        <directory>${basedir}/src/main/groovy</directory>
                        <targetPath>WEB-INF/scripts</targetPath>
                    </resource>
                    <resource>
                        <directory>${prepared.gwt.resources.directory}</directory>
                        <excludes>
                            <exclude>**/*</exclude>
                        </excludes>
                    </resource>
                </webResources>
            </configuration>
        </plugin>
    </plugins>
</build>
